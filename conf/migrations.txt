-- Create tbt_genesis view
-- public.tbt_genesis source

CREATE OR REPLACE VIEW public.tbt_genesis
AS WITH tbt_country AS (
         SELECT sm.run_id::text AS run_id,
                CASE
                    WHEN im.provider::text = ANY (ARRAY['OM-VAL'::character varying, 'OM_CUSTOM'::character varying]::text[]) THEN 'Orbis'::text
                    WHEN im.provider::text = 'TT'::text OR im.provider::text = 'GENESIS'::text THEN 'Genesis'::text
                    WHEN im.provider::text = 'OSM'::text THEN 'OSM'::text
                    ELSE NULL::text
                END AS provider,
            im.country,
            aw.ranking,
            im.product,
            to_char(im.mapdate::date::timestamp with time zone, 'yyyy-mm-01'::text)::date AS mapdate,
            sm.routes,
            round(sm.provider_hours::numeric, 2) AS "time",
            round(sm.provider_km::numeric, 2) AS lenght,
            round(sm.eph::numeric, 3) AS eph,
            round(sm.eph_lower::numeric, 3) AS eph_ci_lower,
            round(sm.eph_upper::numeric, 3) AS eph_ci_upper,
            sm.errors,
            round(((sm.metrics_per_error_type -> 'turnrest'::text) ->> 'eph'::text)::numeric, 3) AS turnrest_eph,
            ((sm.metrics_per_error_type -> 'turnrest'::text) ->> 'errors'::text)::integer AS turnrest_errors,
            round(((sm.metrics_per_error_type -> 'oneway'::text) ->> 'eph'::text)::numeric, 3) AS oneway_eph,
            ((sm.metrics_per_error_type -> 'oneway'::text) ->> 'errors'::text)::integer AS oneway_errors,
            round(((sm.metrics_per_error_type -> 'nonnav'::text) ->> 'eph'::text)::numeric, 3) AS nonnav_eph,
            ((sm.metrics_per_error_type -> 'nonnav'::text) ->> 'errors'::text)::integer AS nonnav_errors,
            date(im.inspection_date) AS inspection_date
           FROM scheduled_metrics sm
             LEFT JOIN inspection_metadata im ON sm.run_id = im.run_id
             LEFT JOIN sampling_metadata sm2 ON im.sample_id = sm2.sample_id
             LEFT JOIN aggregation_weights aw ON im.country::text = aw.country
          WHERE sm.publish_value_stream = true AND (im.provider::text = ANY (ARRAY['TT'::character varying, 'GENESIS'::character varying]::text[])) AND im.mapdate IS NOT NULL AND sm2.metric::text = 'TbT 3.0'::text
        )
 SELECT tbt_country.run_id,
    tbt_country.provider,
    tbt_country.country,
    tbt_country.ranking,
    tbt_country.product,
    tbt_country.mapdate,
    tbt_country.routes,
    tbt_country."time",
    tbt_country.lenght,
    tbt_country.eph,
    tbt_country.eph_ci_lower,
    tbt_country.eph_ci_upper,
    tbt_country.errors,
    tbt_country.turnrest_eph,
    tbt_country.turnrest_errors,
    tbt_country.oneway_eph,
    tbt_country.oneway_errors,
    tbt_country.nonnav_eph,
    tbt_country.nonnav_errors,
    tbt_country.inspection_date
   FROM tbt_country
  WHERE true AND tbt_country.provider = 'Genesis'::text AND tbt_country.country::text <> 'US-CA'::text
  ORDER BY tbt_country.provider, tbt_country.mapdate, tbt_country.ranking;

-- Permissions

ALTER TABLE public.tbt_genesis OWNER TO orbisanalytics;
GRANT ALL ON TABLE public.tbt_genesis TO orbisanalytics;
GRANT SELECT ON TABLE public.tbt_genesis TO genesis_user;

-- create inspection_results view
-- public.inspection_results source

CREATE OR REPLACE VIEW public.inspection_results
AS SELECT im.run_id,
    im.sample_id,
    im.country,
    im.provider,
    im.product,
    im.endpoint,
    im.mcp_tasks,
    im.completed,
    im.mcp_review,
    im.inspection_date,
    im.mcp_end_date,
    im.comment,
    COALESCE(sm2.metric, 'TbT 3 (legacy)'::character varying) AS sample_metric,
    sm2.final_routes AS sample_routes,
    sm2.sample_q,
    sm.routes,
    sm.provider_km,
    sm.provider_hours,
    sm.errors,
    sm.eph,
    sm.eph_lower,
    sm.eph_upper,
    sm.failed_routes,
    sm.failed_routes_lower,
    sm.failed_routes_upper,
    ((im.api_calls -> 'Google'::text)::text)::integer AS google_api_calls,
    ((im.api_calls -> 'Here'::text)::text)::integer AS here_api_calls
   FROM inspection_metadata im
     LEFT JOIN sampling_metadata sm2 USING (sample_id)
     LEFT JOIN scheduled_metrics sm USING (run_id);

-- Permissions

ALTER TABLE public.inspection_results OWNER TO orbisanalytics;
GRANT ALL ON TABLE public.inspection_results TO orbisanalytics;
